table Medicos {
  id uuid [primary key , not null]
  nome varchar [not null]
  endereco varchar 
  telefone varchar [unique, not null]
  especialidade varchar [not null]
}

table Pacientes {
  id uuid [primary key]
  cpf varchar [unique, not null]
  nome varchar [not null]
  endereco varchar [not null]
  telefone varchar [not null]
}

table Dependentes {
  id uuid [primary key]
  id_pacientes uuid [ref :> Pacientes.id]
  nome varchar [not null]
}

table Consultar {
   id uuid [primary key]
   id_medico uuid  [ref: > Medicos.id]
   id_paciente uuid [ref : > Pacientes.id]
   data_consulta date [not null]
   obs varchar
}

table Exames {
  id uuid [primary key]
  id_consultar uuid [ref :> Consultar.id]
  nome varchar [not null]
  cpf_paciente varchar [ref:> Pacientes.cpf]
  resultados varchar  
}
