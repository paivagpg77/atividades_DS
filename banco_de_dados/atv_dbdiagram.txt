table Medicos {
  id uuid [primary key , not null]
  nome varchar [not null]
  endereco varchar 
  telefone varchar [unique, not null]
  especialidade varchar [not null]
}

table Pacientes {
  id uuid [primary key]
  cpf varchar [unique, not null]
  nome varchar
  endereco varchar
  telefone varchar [not null ]
}

table consultas {
   id int [primary key]
   id_medico uuid  [ref: > Medicos.id]
   id_paciente uuid [ref : > Pacientes.id]
   data_consulta date
   exames_passados varchar 
}

table exames {
  id int [primary key]
  id_pacientes varchar [ref : > Pacientes.id]
  id_Medicos varchar [ref :> Medicos.id]
  nome_exame varchar [ref :> consultas.exames_passados]
  resultado_exame varchar
}
